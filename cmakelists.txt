cmake_minimum_required(VERSION 3.22)
project(ref LANGUAGES C CXX)

set(CMAKE_C_STANDARD 17)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# This makes sure that the dynamic library goes into the build directory automatically.
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/$<CONFIGURATION>")
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/$<CONFIGURATION>")

# Gather the source files.
file(GLOB_RECURSE SOURCES
    "src/*.cpp"
    "src/*.c"
)

# Point CMake at the bundled SDL3 package.
list(PREPEND CMAKE_PREFIX_PATH "${CMAKE_CURRENT_SOURCE_DIR}/src/lib/SDL3")
find_package(SDL3 CONFIG REQUIRED)

add_executable(ref ${SOURCES})

target_include_directories(ref PRIVATE
    "${CMAKE_CURRENT_SOURCE_DIR}/src/lib/GLAD/include"
)

target_link_libraries(ref PRIVATE SDL3::SDL3)

if(WIN32)
    add_custom_command(TARGET ref POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
            $<TARGET_RUNTIME_DLLS:ref>
            $<TARGET_FILE_DIR:ref>
        COMMAND_EXPAND_LISTS
        COMMENT "Copy SDL3 runtime DLLs next to the executable")
endif()
